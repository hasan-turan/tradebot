create virtual env: python -m venv path\env-name
activate virtual env: path\env-name\Scripts\activate.bat
install django in virtual env: pip install django

create django project: django-admin startproject pathto-folder(x) . (. for creating project in x folder)

run server on virtual env: python manage.py runserver portnumber

settings.py içinde installed_apps altındaki uygulamalar için gerekli olan veritabanı bilgilerinin oluşturulması için :python manage.py migrate

create dbsuperuser: python manage.py createsuperuser
----> dbusername: hasan
----> pass: 1


start app: python manage.py startapp appname
